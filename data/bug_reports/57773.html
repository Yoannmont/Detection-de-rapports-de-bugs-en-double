/*
 * Created on Oct 14, 2003

package org.atrack.persistence.api;

/**
 * 
 * Interface for ACID transactions that update resources (databases,
 * message queues, transactional file systems, etc.).
 * 
 * aTrack currently uses Hibernate to wrapper JDBC transactions, but this
 *   
 * A transaction instance must have exactly one of commit or rollback
 * called on it exactly once for correct usage. This is handled by the
 * TransactionControl aspect - clients should never call these methods.
 * 
 * @author Ron Bodkin
 *
 */
public interface Transaction {
    /**
	 * 
     * Commit the changes for this transaction (i.e., persist them).
     * May throw an exception if deadlock is detected.
     * 
	 */
	public void commit();
    
    /**
	 * Rollback the changes for this transaction (i.e., abort them). 
	 */
	public void rollback();
    
    
    /**
     * Mark that the transaction is "poisoned." I.e., it will have to rollback.
     * If commit() is called, the transaction will roll back instead.
     * This provides an alternative way of forcing rollback to exiting a
     * transaction control context via an exception.
     *
     */
    public void setRollbackOnly();

    /**
     * 
     * @return whether the transaction will rollback at completion (commit)
     * time.
     */
    public boolean isRollbackOnly();    
}
