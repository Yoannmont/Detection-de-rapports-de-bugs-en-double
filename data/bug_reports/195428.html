Description Juan Barisich 2007-07-04 14:13:07 EDT 
Related with the bug 195414.
When the user is resizing a column of a Table, the TableEditor is redrawed
indefinably. This causes a flashing when is associated a listener to the event
paint of a Table that executes a view action.
My particular case is:
aTable.addListener(SWT.Paint, new Listener() {

      public void handleEvent(Event event) {
        // reset bounds of labels that simulate columns groups titles
      }
    });
This reset of bounds of labels does not behave properly (its flashes) when the
user resize a column (with the mouse) becouse a resize of TableEditor is
executed indefinably at the same time.

The solution (maybe) is:
in the class org.eclipse.swt.custom.ControlEditor
at line 211
the method resize() is:
-----------------------------------------------
void resize () {
  if (editor == null || editor.isDisposed()) return;
  if (editor.getVisible ()) {
    hadFocus = editor.isFocusControl();
  } // this doesn't work because
    // resizing the column takes the focus away
    // before we get here
  editor.setBounds (computeBounds ());
  if (hadFocus) {
    if (editor == null || editor.isDisposed()) return;
    editor.setFocus ();
  }
}
-----------------------------------------------
This must be replace with:
-----------------------------------------------
void resize () {
  if (editor == null || editor.isDisposed()) return;
  if (editor.getVisible ()) {
    hadFocus = editor.isFocusControl();
  } // this doesn't work because
    // resizing the column takes the focus away
    // before we get here
  // THE NEXT LINE FIX THE ERROR
  if (!editor.getBounds().equals(computeBounds())) {
    editor.setBounds (computeBounds ());
  }
  if (hadFocus) {
    if (editor == null || editor.isDisposed()) return;
    editor.setFocus ();
  }
}
-----------------------------------------------
Above, the if condition, avoid the repaints of the table Editor if is
unnecessary. This make the actions associated to the paint event of the table
works ok.

I tell you that I try to simulate a header column group. I improve this with a buttom that is placed above de table, and its preserve the same size of the columns that its contain (or group). Thus, it must have the same width all the time, even when the user is resizing the columns. 

I try to make better my example. Please run it:


import org.eclipse.swt.SWT;
import org.eclipse.swt.custom.TableEditor;
import org.eclipse.swt.events.ModifyEvent;
import org.eclipse.swt.events.ModifyListener;
import org.eclipse.swt.events.PaintEvent;
import org.eclipse.swt.events.PaintListener;
import org.eclipse.swt.events.SelectionAdapter;
import org.eclipse.swt.events.SelectionEvent;
import org.eclipse.swt.layout.GridData;
import org.eclipse.swt.layout.GridLayout;
import org.eclipse.swt.widgets.Button;
import org.eclipse.swt.widgets.Composite;
import org.eclipse.swt.widgets.Control;
import org.eclipse.swt.widgets.Display;
import org.eclipse.swt.widgets.Label;
import org.eclipse.swt.widgets.Shell;
import org.eclipse.swt.widgets.Table;
import org.eclipse.swt.widgets.TableColumn;
import org.eclipse.swt.widgets.TableItem;
import org.eclipse.swt.widgets.Text;

public class ColumnResizeTest {

	public static void main(String[] args) {
		Display display = new Display();
		Shell shell = new Shell(display);
		shell.setLayout(new GridLayout());
		
		final Button button = new Button(shell, SWT.NONE);
		button.setText("Size of Column 1");
		
		final Label labela = new Label(shell, SWT.NONE);
		labela.setText("Click on any item of a table and change the Column 1 width.\nDo the same with both tables, and look how the button is resized.\nIn the first case, the resize is not continuous. The second case its ok");
		
		Composite composite2 = new Composite(shell, SWT.BORDER);
		composite2.setLayout(new GridLayout(2, true));
		composite2.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
		
		final Label label1 = new Label(composite2, SWT.NONE);
		label1.setText("Table 1:");
		
		final Label label2 = new Label(composite2, SWT.NONE);
		label2.setText("Table 2:");
		
		final Table table = new Table(composite2, SWT.FULL_SELECTION
				| SWT.HIDE_SELECTION);
		table.setHeaderVisible(true);
		table.setLinesVisible(true);
		table.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
		
		final TableColumn column1 = new TableColumn(table, SWT.NONE);
		column1.setText("Column 1");
		column1.setWidth(100);
				
		TableColumn column2 = new TableColumn(table, SWT.NONE);
		column2.setText("Column 2");
		for (int i = 0; i < 10; i++) {
			TableItem item = new TableItem(table, SWT.NONE);
			item.setText(new String[] { "item " + i, "edit this value" });
		}
		column2.pack();
		
		final Table table2 = new Table(composite2, SWT.FULL_SELECTION
				| SWT.HIDE_SELECTION);
		table2.setHeaderVisible(true);
		table2.setLinesVisible(true);
		final TableColumn column12 = new TableColumn(table2, SWT.NONE);
		column12.setText("Column 1");
		column12.setWidth(100);
		TableColumn column22 = new TableColumn(table2, SWT.NONE);
		column22.setText("Column 2");
		for (int i = 0; i < 10; i++) {
			TableItem item = new TableItem(table2, SWT.NONE);
			item.setText(new String[] { "item " + i, "edit this value" });
		}
		table2.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
		column22.pack();

		final TableEditor editor = new TableEditor(table);
		// The editor must have the same size as the cell and must
		// not be any smaller than 50 pixels.
		editor.horizontalAlignment = SWT.LEFT;
		editor.grabHorizontal = true;
		editor.minimumWidth = 50;
		// editing the second column
		final int EDITABLECOLUMN = 1;

		table.addPaintListener(new PaintListener() {

			public void paintControl(PaintEvent e) {
				button.setBounds(0, 0, column1.getWidth(),
						button.getBounds().height);

			}

		});

		 table.addSelectionListener(new SelectionAdapter() {
		 public void widgetSelected(SelectionEvent e) {
				// Clean up any previous editor control
				Control oldEditor = editor.getEditor();
				if (oldEditor != null)
					oldEditor.dispose();

				// Identify the selected row
				TableItem item = (TableItem) e.item;
				if (item == null)
					return;

				// The control that will be the editor must be a child of the
				// Table
				Text newEditor = new Text(table, SWT.NONE);
				newEditor.setText(item.getText(EDITABLECOLUMN));
				newEditor.addModifyListener(new ModifyListener() {
					public void modifyText(ModifyEvent me) {
						Text text = (Text) editor.getEditor();
						editor.getItem()
								.setText(EDITABLECOLUMN, text.getText());
					}
		 });
		 newEditor.selectAll();
		 newEditor.setFocus();
		 editor.setEditor(newEditor, item, EDITABLECOLUMN);
		 }
		 });
		 table2.addPaintListener(new PaintListener() {

				public void paintControl(PaintEvent e) {
					button.setBounds(0, 0, column12.getWidth(),
							button.getBounds().height);

				}

			});
		shell.setSize(500, 500);
		shell.open();

		while (!shell.isDisposed()) {
			if (!display.readAndDispatch())
				display.sleep();
		}
		display.dispose();
	}

}
