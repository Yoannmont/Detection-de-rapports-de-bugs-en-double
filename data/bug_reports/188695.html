package de.fuberlin.inf.services.test;

import org.apache.log4j.Logger;
import org.eclipse.birt.chart.device.IDeviceRenderer;
import org.eclipse.birt.chart.exception.ChartException;
import org.eclipse.birt.chart.factory.GeneratedChartState;
import org.eclipse.birt.chart.factory.Generator;
import org.eclipse.birt.chart.model.Chart;
import org.eclipse.birt.chart.model.attribute.Bounds;
import org.eclipse.birt.chart.model.attribute.impl.BoundsImpl;
import org.eclipse.birt.chart.util.PluginSettings;
import org.eclipse.swt.events.PaintEvent;
import org.eclipse.swt.events.PaintListener;
import org.eclipse.swt.graphics.GC;
import org.eclipse.swt.graphics.Image;
import org.eclipse.swt.graphics.Point;
import org.eclipse.swt.graphics.Rectangle;
import org.eclipse.swt.layout.GridData;
import org.eclipse.swt.widgets.Canvas;
import org.eclipse.swt.widgets.Composite;
import org.eclipse.swt.widgets.Display;

public class HistoryChartCanvas extends Canvas implements PaintListener {

	static Logger logger = Logger.getLogger(HistoryChartCanvas.class);

	private IDeviceRenderer deviceRenderer = null;

	private Chart chart = null;

	private Image cachedImage;

	private GeneratedChartState state = null;

	public HistoryChartCanvas(Composite parent, int style) {

		super(parent, style);
		this.setLayoutData(new GridData(GridData.FILL_BOTH));

		// initialize the SWT rendering device
		try {
			PluginSettings ps = PluginSettings.instance();
			deviceRenderer = ps.getDevice("dv.SWT");
		} catch (ChartException pex) {
			logger.error("Couldn't initialize renderer.", pex);
		}

		addPaintListener(this);
		
		logger.debug("Canvas initialized.");
	}
	
	public void paintControl(PaintEvent e) {
		
		logger.debug("Painting...");

		Composite co = (Composite) e.getSource();
		final Rectangle rect = co.getClientArea();

		if (chart != null) {
			if (cachedImage == null) {

				buildChart();
				drawToCachedImage(rect);
			}

			logger.debug("Drawing image...");
			e.gc.drawImage(cachedImage, 0, 0,
					cachedImage.getBounds().width, cachedImage
							.getBounds().height, 0, 0, rect.width,
					rect.height);
		} else
			logger.debug("No chart to draw.");
	}

	private void buildChart() {
		
		logger.debug("Building chart...");

		Point size = getSize();
		Bounds bounds = BoundsImpl.create(0, 0, size.x, size.y);
		int resolution = deviceRenderer.getDisplayServer().getDpiResolution();
		bounds.scale(72d / resolution);
		try {
			Generator gr = Generator.instance();
			state = gr.build(deviceRenderer.getDisplayServer(), chart, bounds,
					null, null, null);
			
			logger.debug("Chart built.");
			
		} catch (ChartException ex) {
			logger.error("Couldn't build chart.", ex);
		}
	}

	public void drawToCachedImage(Rectangle rect) {

		GC gc = null;
		try {
			if (cachedImage != null)
				cachedImage.dispose();
			cachedImage = new Image(Display.getCurrent(), rect);

			gc = new GC(cachedImage);
			deviceRenderer.setProperty(IDeviceRenderer.GRAPHICS_CONTEXT, gc);

			Generator generator = Generator.instance();
			generator.render(deviceRenderer, state);
		} catch (ChartException ex) {
			logger.error("Couldn't create chart.", ex);
		} finally {
			if (gc != null)
				gc.dispose();
		}
	}

	public void update(Chart chart) {

		logger.debug("Updating...");
		this.chart = chart;
		cachedImage = null;
		redraw();
	}
}
Comment 5 Yulin Wang 2007-06-01 02:22:47 EDT 
This bug has been resolved in 2.2, but the problem exactly exists in 2.1.2.
The issue is caused by the fact that the gap is too small and there are double precision error when comparing two ticks.

A workaround of avoid freezing in 2.1.2 is to remove this piece of code: 
"xAxis.getOrigin().setType(IntersectionType.VALUE_LITERAL);"
This script will make X axis starting with 0, but the minimum is 340 and maximum is 346. If removing the code, X axis will start with the minimum 340 by default. This not only could avoid freezing, but also could make Charts look better.

However, fixing this bug with high quality has high risks. I suggest fixing it in 2.2.
