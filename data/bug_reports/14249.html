import org.eclipse.swt.SWT;
import org.eclipse.swt.widgets.*;


/* simplest code that show the problem, describled on PR n# 14249  */
public class TreeTest {
  public static void main(String[] args) {
    Display display = new Display(); 		
    Shell shell = new Shell (display);
    shell.setBounds(30, 30, 300, 300);
    shell.open();
    Tree tree = new Tree( shell, SWT.MULTI | SWT.BORDER );
    tree.setBounds( 20, 20, 200, 200);
    for (int i = 0; i < 4; i++) {
      TreeItem item = new TreeItem( tree, 0);
      item.setText("Item");
    }
    tree.getItems()[1].setForeground( display.getSystemColor(SWT.COLOR_RED));
    tree.dispose(); //NPE
	
    while (! shell.isDisposed() )
      if ( !display.readAndDispatch() )
        display.sleep();
		
    display.dispose();
  }
}
