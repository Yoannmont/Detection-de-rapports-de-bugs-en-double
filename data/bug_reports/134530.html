Comment 8 Steve Northover 2006-04-13 14:35:33 EDT 
Here is some code that shows the problem only happens when the drop down tool item is the first on the tool bar:

public class PR_134530 {
	public static void main(String[] args) {
		Display display = new Display();
		Image icon = new Image(display, 16, 16);
		Shell shell = new Shell(display);
		RowLayout layout = new RowLayout(SWT.VERTICAL);
		shell.setLayout(layout);
		ToolBar toolBar0 = new ToolBar(shell, SWT.FLAT | SWT.BORDER);
		for (int i = 0; i < 8; i++) {
			ToolItem item = new ToolItem(toolBar0, i != 0 ? SWT.PUSH : SWT.DROP_DOWN);
			item.setImage(icon);
			//item.setText("Item" + i);
		}
		toolBar0.pack();
		ToolBar toolBar1 = new ToolBar(shell, SWT.FLAT | SWT.BORDER);
		for (int i = 0; i < 8; i++) {
			ToolItem item = new ToolItem(toolBar1, i != 7 ? SWT.PUSH : SWT.DROP_DOWN);
			item.setImage(icon);
			//item.setText("Item" + i);
		}
		toolBar1.pack();
		shell.pack();
		shell.open();
		while (!shell.isDisposed()) {
			if (!display.readAndDispatch()) display.sleep();
		}
		display.dispose();
	}
}
