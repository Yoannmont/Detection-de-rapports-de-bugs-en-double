Attached is an x86-64 GTK+ application compiled on our build machine.  Below is the source code.  Can you please run this app and see if it says "success" or if it segfaults, and also try compiling the source code and see if that segfaults.


#include <stdio.h>
#include <dlfcn.h>
#include <stdint.h>
#include <gtk/gtk.h>

typedef int64_t (*FPTR)(unsigned char *, int64_t, int32_t, int64_t,
                        int32_t, int64_t, int32_t, int64_t, ...);

void open_chooser (char *arg0, int64_t arg1, int32_t arg2,
                   int64_t arg3, int32_t arg4, int64_t arg5,
                   int32_t arg6, int64_t arg7)
{
    void *handle = dlopen("libgtk-x11-2.0.so.0", RTLD_LAZY);
    FPTR fptr = dlsym(handle, "gtk_file_chooser_dialog_new");
    (fptr)(arg0, arg1, arg2, arg3, arg4, arg5, arg6, arg7);
}

int main (int argc, char **argv)
{
    int zero = 0;
    g_thread_init (0);
    gtk_set_locale ();
    gtk_init (&zero, 0);
    open_chooser ("title", 0, 0,
                  (int64_t) GTK_STOCK_CANCEL, GTK_RESPONSE_CANCEL,
                  (int64_t) GTK_STOCK_OK, GTK_RESPONSE_OK, 0);
    fprintf (stdout, "success\n");
    return 0;
}
