  public TreeColumn addColumn(String text, int style, int width)
  {
    TreeColumn column = new TreeColumn(tree, style);
    column.setText(text);
    column.setWidth(width);

    return column;
  }

  public TreeItem putTreeItem(String uuid, String[] rowData)
  {
    TreeItem item = new TreeItem(tree, SWT.NONE);
    item.setText(rowData);
    item.setData("UUID", uuid);
    return item;
  }

  public TreeItem putTreeItem(TreeItem parent, String uuid, String[] rowData)
  {
    TreeItem item = new TreeItem(parent, SWT.NONE);
    item.setText(rowData);
    item.setData("UUID", uuid);
    return item;
  }

  private String[] buildCaseItem(String caseId, String fees, String balance)
  {
    return new String[] { caseId, "", "", fees, balance};
  }

  private String[] buildCitationItem(String citationId, String defendant, String
fees, String balance)
  {
    return new String[] { citationId + spaces(tree, 30) + defendant, "", "", 
fees, balance};
  }
  
  private String[] buildFeeItem(String feeDesc, String fees, String balance)
  {
    return new String[] { spaces(tree, 160)+feeDesc, "", "",  fees, balance};
  }

  private String[] buildVioliationItem(String count, String vioDesc, String plea,
                                       String disp, String fees, String balance)
  {
    return new String[] {count+spaces(tree,30)+vioDesc, plea, disp, fees, balance};
  }

  public String spaces(Control control, int pixels)
  {
    String spaceStr = "";
    int oneSpace = getStringWidth(control, " ");
    for (int i = 0; i < pixels; i = i + oneSpace)
      spaceStr = spaceStr + " ";
    return spaceStr;
  }

  public int getStringWidth(Control control, String str)
  {
    TextLayout layout = new TextLayout(Display.getDefault());
    layout.setFont(control.getFont());
    layout.setText(str);
    return layout.getBounds().width;
  }

  public TreeEditor buildComboEditor(TreeItem treeItem, EventListener[]
listeners, int column, String[] comboItems)
  {
    final TreeEditor editor = new TreeEditor(tree);
    final CCombo combo = new CCombo(tree, SWT.NONE);

    if (treeItem == null)
      return null;
    
    combo.setItems(comboItems);
    
    if (treeItem.getData() == null)
      treeItem.setData(new Control[tree.getColumnCount()]);
    
    ((Control[])treeItem.getData())[column] = combo;
    combo.setData(treeItem);
    combo.setData("column",column);
    
    if (listeners != null)
    for (EventListener listener : listeners) {
      if (listener != null) {
        if (listener instanceof SelectionListener)
          combo.addSelectionListener((SelectionListener)listener);
        else if (listener instanceof FocusListener)
          combo.addFocusListener((FocusListener)listener);
        else if (listener instanceof ModifyListener)
          combo.addModifyListener((ModifyListener)listener);
      }
    }
    
    treeItem.addDisposeListener(new DisposeListener() {
      public void widgetDisposed( DisposeEvent e )
      {
        combo.dispose();
        editor.dispose();
      }     
    });
    
    editor.grabHorizontal = true;
    editor.grabVertical = true;
    editor.setEditor(combo, treeItem, column);
    combo.setText(treeItem.getText(column));
    
    return editor;
  }
