Comment 31 Veronika Irvine 2005-04-25 13:36:56 EDT 
The following simple example demonstrates the problem.  If the Toolbar is a 
child of a Group or a CoolBar, it draws the correct background.  If it is a 
child of a Composite, it does not.  Note: you need to have a manifest to see 
the problem and you need to be running with the Silver theme.

public static void main (String [] args) {
	Display display = new Display ();
	Shell shell = new Shell (display);
	Composite parent = new Composite(shell, SWT.BORDER);
	//Group parent = new Group(shell, SWT.BORDER); // works for a Group
	ToolBar toolbar = new ToolBar(parent, SWT.FLAT);
	ToolItem item = new ToolItem(toolbar, SWT.NONE);
	item.setText("hello");
	item = new ToolItem(toolbar, SWT.NONE);
	item.setText("world");
	toolbar.pack();
	parent.setSize(400, 200);
	shell.pack();
	shell.open ();
	while (!shell.isDisposed ()) {
		if (!display.readAndDispatch ()) display.sleep ();
	}
	display.dispose ();
}
