Comment 3 Paul Webster 2009-10-19 14:28:49 EDT 
Used this snippet to add the contribution from the activator:

		new UIJob("AddThatMenu") {

			@Override
			public IStatus runInUIThread(IProgressMonitor monitor) {
				System.out.println("Add contribution");
				IMenuService ms = (IMenuService) PlatformUI.getWorkbench()
						.getService(IMenuService.class);
				ms.addContributionFactory(new AbstractContributionFactory(
						"menu:org.eclipse.ui.main.menu?after=file",
						"z.ex.bug256073") {
					@Override
					public void createContributionItems(
							IServiceLocator serviceLocator,
							IContributionRoot additions) {
						MenuManager manager = new MenuManager("&LoFile",
								"lofile");
						CommandContributionItem cci = new CommandContributionItem(
								new CommandContributionItemParameter(
										serviceLocator, "my.about",
										IWorkbenchCommandConstants.HELP_ABOUT,
										CommandContributionItem.STYLE_PUSH));
						manager.add(cci);
						additions.addContributionItem(manager, null);
					}
				});
				monitor.done();
				return Status.OK_STATUS;
			}
		}.schedule();

IViewPart problems = window.showView(IPageLayout.ID_PROBLEM_VIEW);
MenuManager manager = new MenuManager();
manager.setRemoveAllWhenShown(true);
Menu context = manager.createContextMenu(some widget/shell);

PopupMenuExtender popup = new PopupMenuExtender(IPageLayout.ID_PROBLEM_VIEW, manager, problems.getSite().getSelectionProvider(), problems, false);
popup.addMenuItem(MarkerSupportRegistry.MARKERS_ID);

then when you Show/Hide on the context menu, the PopupMenuExtender will add/remove the contribution items from the command framework.
