Description Jeppe Cramon 2003-04-14 07:53:50 EDT 
Running this little test app under both Eclipse 2.0.1 and 2.1 using the VM 
argument -XrunHeapProfile gives an abnormal result when running under Eclipse 
as compared to when it's running from the command line.

Code:
package com.brf.test;

public class TestApp {

	public static void main(String[] args) throws InterruptedException {
		System.out.println("Starting up");
		System.out.println("Going to sleep");
		Thread.sleep(1000);
		System.out.println("Waking up");
		System.out.println("Closing down");
	}
}

The HeapProfile is a JVMPI profiler which has been "ported" to Windows (Visual 
C++ ver. 6)
The original code can be found at: http://www.virtualmachine.de
The complete code is: 
/*

  Copyright (c) 2003, Matthias Ernst <matthias.ernst@coremedia.com>, All 
rights reserved.

  INSTALLING AND RUNNING HEAPPROFILE

  Compiling / Running:

    Linux / Solaris:

      1) gcc -I $JAVA_HOME/include -I $JAVA_HOME/include/linux -fPIC -c 
heapprofile.cpp
      2) gcc -shared -o libheapprofile.so heapprofile.o
      3) LD_LIBRARY_PATH=<path to lib>
      4) java -Xrunheapprofile ...


    OS/X (thanks to Robert Lee <crazybob@crazybob.org>)

      1) Replace "#include <malloc.h>" with "#include <stdlib.h>".
      2) c++ -I $JAVA_HOME/include -I -fno-common -fPIC -c heapprofile.cpp
      3) c++ -dynamic -dynamiclib -o libheapprofile.jnilib heapprofile.o
      4) Add library path to DYLD_LIBRARY_PATH.
      5) Run with -Xrunheapprofile option.


  Dumping:

    kill -QUIT <pid>
    or CTRL-\


  Largest objects:

    sort -n heap.dump | tail -100


  Instance count per class (disregarding size):

    cut -d ' ' -f 2 <heap.dump | sort -n | uniq -c | sort -n



  Matthias Ernst
  matthias.ernst@coremedia.com

*/

#define WIN32_LEAN_AND_MEAN
#include <windows.h>
#include <string.h>
#include <malloc.h>
//#include <unistd.h>
#include <conio.h>
#include "jvmpi.h"


// global jvmpi interface pointer
static JVMPI_Interface *jvmpi_interface;
static jobjectID object;
static jobjectID clazz;
static int objectsize;
static const char *classname;
char *buf = (char *)malloc(5);

static jobjectID clazzcache;
static const char *namecache;

void load_class() {
  jvmpi_interface->RequestEvent(JVMPI_EVENT_OBJECT_ALLOC, object);
}

void load_classname() {
  if(clazz != clazzcache) {
    jvmpi_interface->RequestEvent(JVMPI_EVENT_CLASS_LOAD, clazz);
    clazzcache = clazz;
    namecache = classname;
  }
}

jobjectID asObject(char *curr) {
  char ptr[4];
  memcpy(ptr, curr, 4);
  return *((jobjectID*)ptr);
}

// function for handling event notification
void notifyEvent(JVMPI_Event *event) {
  switch(event->event_type) {
  case JVMPI_EVENT_CLASS_LOAD  | JVMPI_REQUESTED_EVENT:
    classname = event->u.class_load.class_name;
    break;
  case JVMPI_EVENT_OBJECT_ALLOC  | JVMPI_REQUESTED_EVENT:
    objectsize = event->u.obj_alloc.size;
    clazz = event->u.obj_alloc.class_id;
    if(clazz == NULL) {
      if(event->u.obj_alloc.is_array != JVMPI_NORMAL_OBJECT) {
        sprintf(buf, "[%d", event->u.obj_alloc.is_array);
	classname = buf;
      }
      else
        classname = "NULL?";
    } else {
      load_classname();
    }
    break;
  case JVMPI_EVENT_DATA_DUMP_REQUEST:
  case JVMPI_EVENT_JVM_SHUT_DOWN:
    jvmpi_interface->RunGC();
    Sleep(5);
    jvmpi_interface->RunGC();
    JVMPI_HeapDumpArg arg;
    arg.heap_dump_level = JVMPI_DUMP_LEVEL_0;
    jvmpi_interface->RequestEvent(JVMPI_EVENT_HEAP_DUMP, &arg);
    break;
  case JVMPI_EVENT_HEAP_DUMP | JVMPI_REQUESTED_EVENT: { 
    FILE *f = fopen("heap.dump", "w");
    if(f == NULL) return;

    int count = 0;
    char *curr = event->u.heap_dump.begin;
    while(curr < event->u.heap_dump.end) {
      if(((++count)%1000)==0) fprintf(stderr, ".");
      int type = *(curr++);
      object = asObject(curr);
      if(object == NULL) fprintf(stderr, "object null\n");
      curr+=4;
      load_class();
      fprintf(f, "%d, %s\n", objectsize, classname);
    }

    fclose(f);
    break;
  }
  }
}

// profiler agent entry point
extern "C" { 
  JNIEXPORT jint JNICALL JVM_OnLoad(JavaVM *jvm, char *options, void 
*reserved) {
    fprintf(stderr, "myprofiler> initializing ..... \n");
    
    // get jvmpi interface pointer
    if ((jvm->GetEnv((void **)&jvmpi_interface, JVMPI_VERSION_1)) < 0) {
      fprintf(stderr, "myprofiler> error in obtaining jvmpi interface 
pointer\n");
      return JNI_ERR;
    } 
    
    jvmpi_interface->NotifyEvent = notifyEvent;
    jvmpi_interface->EnableEvent(JVMPI_EVENT_DATA_DUMP_REQUEST, NULL);
    jvmpi_interface->EnableEvent(JVMPI_EVENT_JVM_SHUT_DOWN, NULL);

    fprintf(stderr, "heapprofile .... running \n\n");
    return JNI_OK;
  }
}

When running under Eclipse using both Sun's JDK 1.4.1_02 (under Eclipse 2.01 
and Eclipse 2.1) and the IBM JRE from WSAD 5 (Eclipse 2.01) as execution JRE, 
the result looks like this:
Starting up
Going to sleep
Waking up
Closing down
myprofiler> initializing ..... 
heapprofile .... running 

...

When running it from the command line (using Suns JDK 1.4.1_02) it looks like 
this:
myprofiler> initializing .....
heapprofile .... running

Starting up
Going to sleep
Waking up
Closing down
...

So when it's running under Eclipse the profiler isn't loaded until the main 
method exists... Try change the sleep routine to an endless loop and you will 
never see any profiling text.

Also, I'm missing a way/method under Eclipse to active the profiling dump 
(Ctrl-Break) to send the JVMPI_EVENT_DATA_DUMP_REQUEST event.

/Jeppe

Comment 2 Darin Wright 2003-04-21 13:38:10 EDT 
Ctrl-Break is not supported in the Eclipse console. The workaround is to launch 
java programs using "java" instead of "javaw" (from the JRE options in the JRE 
tab of the launch config). This will produce a DOS console when you launch your 
program, where you can use Ctrl-break. Note, for this to work, you must launch 
Eclipse without a DOS console.

Comment 10 Michael Rennie 2007-06-25 15:33:52 EDT 
Created attachment 72394 [details]
modified source file

The problem comes from the nature of windows output streams, where it seems the stderr stream is a little more lazy in flushing than the stdout stream. 

If we flush the stderr stream (like we always should be doing as per good programming practices) the output to the console in Eclipse is correct. The underlying issue here is how the consoles behave for the specific system stderr/stdout streams. The Eclipse console constantly tries to read from any output streams it has attached to and will print out bytes from the stream as the stream reports they are available, so if the stream does not indicate it has bytes to read, the console blocks on trying to read until more bytes are available.

Thats why flushing stderr works, we are forcing the stream to say it has more bytes, instead of waiting for the process to terminate forcing a flush of the attached stderr stream (in effect the same behavior, but the reason it would always appear at the end when running the program).
