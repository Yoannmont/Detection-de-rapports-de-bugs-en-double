Comment 13 Grant Gayed 2004-06-03 13:28:14 EDT 
The following snippet shows what appears to be the root cause.  When tree item 
is deselected, right-clicking on it fires Tree.selection and then Menu.show on 
win32, but reverses the order of these events on motif.  The menu management 
magic above us seems to depend on the win32 event ordering, since the 
Tree.selection event is ultimately used to determine the proper context menu 
items to create when Menu.show is received.

public static void main(String[] args) {
	final Display display = new Display ();
	Shell shell = new Shell(display);
	shell.setBounds(10,10,200,200);
	Tree tree = new Tree(shell, SWT.MULTI);
	tree.setBounds(10,10,100,100);
	new TreeItem(tree, SWT.NONE).setText("tree item");
	Menu menu = new Menu (tree);
	MenuItem item = new MenuItem (menu, SWT.PUSH);
	item.setText ("menu item");
	tree.setMenu (menu);
	tree.addListener(SWT.Selection, new Listener() {
		public void handleEvent(Event event) {
			System.out.println("tree selection");
		}
	});
	menu.addListener(SWT.Show, new Listener() {
		public void handleEvent(Event event) {
			System.out.println("menu show");
		}
	});
	shell.open();
	while (!shell.isDisposed()) {
		if (!display.readAndDispatch()) display.sleep();
	}
	display.dispose();
}
