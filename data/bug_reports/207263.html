Comment 10 Steve Northover 2007-10-25 15:07:18 EDT 
Fixed > 20071025

The NPE is gone but there is an issue with Windows.  Setting a submenu requires that the menu item parent (the cascade) be destroyed and recreated in the operating system.  There is no way around this, other than changing the UI code to make use of this by first disposing the old menu then setting the new.  Here is some example code that shows the work around:

import org.eclipse.swt.*;
import org.eclipse.swt.widgets.*;

public class PR_207263 {
	public static void main (String [] args) {
		final Display display = new Display ();
		Shell shell = new Shell (display);
		Menu menu = new Menu (shell, SWT.POP_UP);
		final MenuItem item = new MenuItem (menu, SWT.CASCADE);
		item.setText ("Cascade Item");
		MenuItem item3 = new MenuItem (menu, SWT.PUSH);
		item3.setText ("Another Item");
		
		final Menu menu1 = new Menu (shell, SWT.DROP_DOWN);
		MenuItem item1 = new MenuItem (menu1, SWT.PUSH);
		item1.setText ("Old Item");
		final Menu menu2 = new Menu (shell, SWT.DROP_DOWN);
		MenuItem item2 = new MenuItem (menu2, SWT.PUSH);
		item2.setText ("New Item1");
		item.setMenu (menu1);
		menu1.addListener (SWT.Hide, new Listener () {
			public void handleEvent (Event e) {
				menu1.removeListener (SWT.Hide, this);
				e.display.asyncExec (new Runnable () {
					public void run () {
						/* Dispose the old menu before setting the new one */
						menu1.dispose ();
						item.setMenu (menu2);
					}
				});
			}
		});
		
		shell.setMenu (menu);
		shell.pack ();
		shell.open ();
		while (!shell.isDisposed ()) {
			if (!display.readAndDispatch ()) display.sleep ();
		}
		display.dispose ();
	}
}
