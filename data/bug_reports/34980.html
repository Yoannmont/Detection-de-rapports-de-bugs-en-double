Tree and Table widgets. So if the images are PNG with alpha info,
they will probably show the transparent areas in black.

Comment 8 Silenio Quarti 2003-04-28 13:00:23 EDT 
The tree/table/list wigets were written in Eclipse 2.1. We started
using the new GTK widget GtkTreeView.

We do not have a machine to test/reproduce your problem. Please could
you make the following change to org.eclipse.swt.widgets.ImageList and
see if the problem goes way. The code is suspicious since PPC is big
endian.

public int add (Image image) {
	int index = 0;
	while (index < images.length) {
		if (images [index] != null) {
			if (images [index].isDisposed ()) {
				OS.g_object_unref (pixbufs [index]);
				images [index] = null;
				pixbufs [index] = 0;
			}
		}
		if (images [index] == null) break;
		index++;
	}
	int [] w = new int [1], h = new int [1];
 	OS.gdk_drawable_get_size (image.pixmap, w, h);
	int width = w [0], height = h [0]; 	
	boolean hasMask = image.mask != 0;
	int pixbuf = OS.gdk_pixbuf_new (OS.GDK_COLORSPACE_RGB, hasMask, 8, 
width, height);
	if (pixbuf == 0) SWT.error (SWT.ERROR_NO_HANDLES);
	int colormap = OS.gdk_colormap_get_system ();
	OS.gdk_pixbuf_get_from_drawable (pixbuf, image.pixmap, colormap, 0, 0, 
0, 0, width, height);
	if (hasMask) {
		int gdkMaskImagePtr = OS.gdk_drawable_get_image (image.mask, 0, 
0, width, height);
		if (gdkMaskImagePtr == 0) SWT.error (SWT.ERROR_NO_HANDLES);
		int stride = OS.gdk_pixbuf_get_rowstride (pixbuf);
		int pixels = OS.gdk_pixbuf_get_pixels (pixbuf);
		byte [] line = new byte [stride];
		for (int y=0; y<height; y++) {
			int offset = pixels + (y * stride);
			OS.memmove (line, offset, stride);
			for (int x=0; x<width; x++) {
				if (OS.gdk_image_get_pixel (gdkMaskImagePtr, x, 
y) != 0) {
					line[x*4] = (byte)0xFF;
				} else {
					line[x*4] = 0;
				}
			}
			OS.memmove (offset, line, stride);
		}
		OS.g_object_unref (gdkMaskImagePtr);
	}
	if (index == images.length) {
		Image [] newImages = new Image [images.length + 4];
		System.arraycopy (images, 0, newImages, 0, images.length);
		images = newImages;
		int [] newPixbufs = new int [pixbufs.length + 4];
		System.arraycopy (pixbufs, 0, newPixbufs, 0, pixbufs.length);
		pixbufs = newPixbufs;
	}
	pixbufs [index] = pixbuf;
	images [index] = image;
	return index;
}
