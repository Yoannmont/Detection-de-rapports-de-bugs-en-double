Not being very familiar with the code I tried a hack that would compare the deleted code to the changed code and if they are identical (except for white space) then skip all the parsing:

    private StructuredDocumentEvent checkForQuotes() {
        // routine is supported with null or empty string meaning the same
        // thing: deletion
        if (fChanges == null)
            fChanges = ""; //$NON-NLS-1$
        //
        StructuredDocumentEvent result = null;
        try {
            if ((fChanges != null) && (fDeletedText != null)) {
                boolean same = true;
                int fChangesIndex = 0;
                int fDeletedTextIndex = 0;
                while((fChangesIndex < fChanges.length()) && (fDeletedTextIndex < fDeletedText.length())) {
                    if (fChangesIndex < fChanges.length()) {
                        if (Character.isWhitespace(fChanges.charAt(fChangesIndex))) {
                            fChangesIndex++;
                            continue;
                        }
                    }
                    if (fDeletedTextIndex < fDeletedText.length()) {
                        if (Character.isWhitespace(fDeletedText.charAt(fDeletedTextIndex))) {
                            fDeletedTextIndex++;
                            continue;
                        }
                    }
                    if ((fChangesIndex >= fChanges.length()) && fDeletedTextIndex < fDeletedText.length()) {
                        same = false;
                        break;
                    }
                    if ((fChangesIndex < fChanges.length()) && fDeletedTextIndex >= fDeletedText.length()) {
                        same = false;
                        break;
                    }
                    if (fChanges.charAt(fChangesIndex) != fDeletedText.charAt(fDeletedTextIndex)) {
                        same = false;
                        break;
                    } else {
                        fChangesIndex++;
                        fDeletedTextIndex++;
                    }
                }
                if (same) {
                    return result;
                }
            }
