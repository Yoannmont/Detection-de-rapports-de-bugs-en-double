Description Mark Mahieu 2006-05-01 10:48:00 EDT 
SWT 3.2rc2, Windows XP sp2

Tables with the SWT.VIRTUAL style set cause MeasureItem, PaintItem and EraseItem listeners to fire repeatedly every time the mouse moves over the TableItems.  Example code below: run it and move the mouse around over the Table (and watch the console output).

This does not seem to occur with non-virtual Tables.  I haven't noticed it on virtual Trees either.

On the machine I'm using now (2GHz P4, integrated graphics) this is causing a distracting amount of flicker, as the TableItems are continually erased and repainted.


import org.eclipse.swt.*;
import org.eclipse.swt.layout.*;
import org.eclipse.swt.widgets.*;


public class VirtualTable {

	static int eventCount = 0;

	public static void main(String[] args) {

		final Display display = new Display();
		Shell shell = new Shell(display);
		shell.setLayout(new FillLayout());
		
		final Table table = new Table(shell, SWT.VIRTUAL);
		table.setLinesVisible(true);
		new TableColumn(table, SWT.NONE).setWidth(200);
		table.setItemCount(40);

		Listener listener = new Listener() {

			public void handleEvent(Event event) {

				switch (event.type) {
					case SWT.SetData: {
						System.out.println(eventCount++ + " SetData");
						break;
					}
					case SWT.MeasureItem: {
						System.out.println(eventCount++ + " MeasureItem");
						break;
					}
					case SWT.PaintItem: {
						System.out.println(eventCount++ + " PaintItem");
						break;
					}
					case SWT.EraseItem: {
						System.out.println(eventCount++ + " EraseItem");
						break;
					}
				}
			}
		};
		
		table.addListener(SWT.SetData, listener);
		table.addListener(SWT.MeasureItem, listener);
		table.addListener(SWT.PaintItem, listener);
		table.addListener(SWT.EraseItem, listener);

		shell.setSize(250, 300);
		shell.open();
		
		while (!shell.isDisposed()) {
			if (!display.readAndDispatch()) {
				display.sleep();
			}
		}
		
		display.dispose();
	}
}

(In reply to comment #3)

Sorry I should have been clearer - the flicker I was seeing was with a more complex example, which I'm having trouble simplifying enough to make a useful test case.  The earlier example was merely to demonstrate the amount of calls to the Listeners.

However, I've now managed to get a slightly modified version of Snippet228 to produce the problem.  It only seems to occur on the first column, so as well as making it use a virtual Table I've swapped the columns.  I've also increased the size of the TableItems, and the amount of text drawn in the PaintItem Listener, which makes the effect more obvious (on my machine):


import org.eclipse.swt.*;
import org.eclipse.swt.graphics.*;
import org.eclipse.swt.layout.*;
import org.eclipse.swt.widgets.*;

public class Snippet228Flicker {
		
	public static void main(String [] args) {
		final Display display = new Display();		
		Shell shell = new Shell(display);
		shell.setLayout(new FillLayout());
		shell.setText("Show results as a bar chart in Table");
		final Table table = new Table(shell, SWT.VIRTUAL | SWT.BORDER);
		table.setHeaderVisible(true);
		table.setLinesVisible(true);
		TableColumn column1 = new TableColumn(table, SWT.NONE);
		column1.setText("Bug Status");
		column1.setWidth(400);
		final TableColumn column2 = new TableColumn(table, SWT.NONE);
		column2.setText("Percent");
		column2.setWidth(200);
		final String[] labels = new String[]{"Resolved", "New", "Won't Fix", "Invalid"};
		
		table.addListener(SWT.SetData, new Listener() {
			public void handleEvent(Event event) {
				TableItem item = (TableItem) event.item;
				int i = table.indexOf(item);
				item.setText(1, labels[i]);
			}
		});
		
		table.addListener(SWT.MeasureItem, new Listener() {
			public void handleEvent(Event event) {
				event.height = 100;
			}
		});
		
		/*
		 * NOTE: MeasureItem, PaintItem and EraseItem are called repeatedly.
		 * Therefore, it is critical for performance that these methods be
		 * as efficient as possible.
		 */
		table.addListener(SWT.PaintItem, new Listener() {
			int[] percents = new int[] {50, 30, 5, 15};
			public void handleEvent(Event event) {
				if (event.index == 0) {
					GC gc = event.gc;
					TableItem item = (TableItem)event.item;
					int index = table.indexOf(item);
					int percent = percents[index];
					Color foreground = gc.getForeground();
					Color background = gc.getBackground();
					gc.setForeground(display.getSystemColor(SWT.COLOR_RED));
					gc.setBackground(display.getSystemColor(SWT.COLOR_YELLOW));
					int width = (column2.getWidth() - 1) * percent / 100;
					gc.fillGradientRectangle(event.x, event.y, width, event.height, true);					
					Rectangle rect2 = new Rectangle(event.x, event.y, width-1, event.height-1);
					gc.drawRectangle(rect2);
					gc.setForeground(display.getSystemColor(SWT.COLOR_LIST_FOREGROUND));
					String text = percent+"% blah blah blah blah blah blah blah blah blah";
					Point size = event.gc.textExtent(text);					
					int offset = Math.max(0, (event.height - size.y) / 2);
					gc.drawText(text, event.x+2, event.y+offset, true);
					gc.setForeground(background);
					gc.setBackground(foreground);
				}
			}
		});		
		
		table.setItemCount(labels.length);
				
		shell.setSize(500, 500);
		shell.open();
		while(!shell.isDisposed()) {
			if(!display.readAndDispatch()) display.sleep();
		}
		display.dispose();
	}
}
Comment 5 Mark Mahieu 2006-05-01 15:08:57 EDT 
I've just tried the same modification with Snippet229 (making it virtual), and I'm seeing something a bit different now - but much more obvious.  Select a row, then move the mouse, and on my machine the first column of the selected row is repainted several times very quickly (becoming plain red due to the incremental alpha paints), overwriting the text.

My understanding of the way this is supposed to work is probably at fault here, but it seems wrong that just switching a table to be virtual should cause such radically different behaviour.

Here's the code:


import org.eclipse.swt.*;
import org.eclipse.swt.graphics.*;
import org.eclipse.swt.layout.*;
import org.eclipse.swt.widgets.*;

public class Snippet229Mod {
	public static void main(String [] args) {
		final Display display = new Display();
		Shell shell = new Shell(display);
		shell.setText("Custom gradient selection for Table");
		shell.setLayout(new FillLayout());
		final Table table = new Table(shell, SWT.VIRTUAL | SWT.MULTI | SWT.FULL_SELECTION);
		table.setHeaderVisible(true);
		table.setLinesVisible(true);
		int columnCount = 3;
		for (int i=0; i<columnCount; i++) {
			TableColumn column = new TableColumn(table, SWT.NONE);
			column.setText("Column " + i);	
		}
		int itemCount = 8;
	
		table.addListener(SWT.SetData, new Listener() { 
			public void handleEvent(Event event) {
				TableItem item = (TableItem) event.item;
				int i = table.indexOf(item);
				item.setText(new String[] {"item "+i+" a", "item "+i+" b", "item "+i+" c"});
			}
		});
		
		/*
		 * NOTE: MeasureItem, PaintItem and EraseItem are called repeatedly.
		 * Therefore, it is critical for performance that these methods be
		 * as efficient as possible.
		 */
		table.addListener(SWT.EraseItem, new Listener() {
			public void handleEvent(Event event) {			
				if((event.detail & SWT.SELECTED) != 0) {
					GC gc = event.gc;
					Rectangle area = table.getClientArea();
					/*
					 * If you wish to paint the selection beyond the end of
					 * last column, you must change the clipping region.
					 */
					int columnCount = table.getColumnCount();
					if (event.index == columnCount - 1 || columnCount == 0) {
						int width = area.x + area.width - event.x;
						if (width > 0) {
							Region region = new Region();
							gc.getClipping(region);
							region.add(event.x, event.y, width, event.height); 
							gc.setClipping(region);
							region.dispose();
						}
					}
					gc.setAdvanced(true);
					if (gc.getAdvanced()) gc.setAlpha(127);								
					Rectangle rect = event.getBounds();
					Color foreground = gc.getForeground();
					Color background = gc.getBackground();
					gc.setForeground(display.getSystemColor(SWT.COLOR_RED));
					gc.setBackground(display.getSystemColor(SWT.COLOR_LIST_BACKGROUND));
					gc.fillGradientRectangle(0, rect.y, 500, rect.height, false);
					// restore colors for subsequent drawing
					gc.setForeground(foreground);
					gc.setBackground(background);
					event.detail &= ~SWT.SELECTED;					
				}						
			}
		});		
		for (int i=0; i<columnCount; i++) {
			table.getColumn(i).pack();
		}	
		
		table.setItemCount(itemCount);
		
		table.setSelection(table.getItem(0));
		shell.setSize(500, 200);
		shell.open();
		while (!shell.isDisposed()) {
			if (!display.readAndDispatch()) display.sleep();
		}
		display.dispose();	
	}
}
